
//Take u forward - striver explanation

class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        //Take u forward
 
        
        //Step 1: //point the last node to head, create continuous connection
        //Step 2: //in case k is divisible by length, return head
        //Step 3: //find kth node, 
        //Step 4: point head to kth next i.e curr.next, break curr i.e. curr.next = null
        //Step 5: return head
        
        //edge cases
        if(head==null ||head.next==null || k==0)
            return head;
        
        //compute the length
        ListNode cur = head;
        int len=1;
        while(cur.next !=null){
            len++;
            cur = cur.next;
        }

        //0
        //point the last node to head, create continuous connection
        cur.next = head;

        //1
        //in case k is divisible by length, return head
        k = k % len;
        if(k == 0) return head

        //2
        //find kth node
        k = len - k;
        // keep traversing len-k to reach the new last node
        while(k-- >0) cur = cur.next;

        //3
        //make kth next as head and break connection by pointing k to null
        head = cur.next; //new head -will be next of cur
        cur.next = null; //assigning null to last node
            
        return head;
    }
}

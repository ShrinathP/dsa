//IMP - prev.next = prev.next.next
// Delete kth node of LL
// Striver explanation and code
1. Hold prev = null, temp = head, cnt = 0 - (need 2 pointers - prev and current)
2. while temp != null, increaase cnt++, move prev = temp, temp = temp.next
3. if cnt = k, move prev.next = prev.next.next

private static Node removeK(Node head, int k) {

    if (head == null) return head;

    //if k=1 we need to remove the head, return head.next
    if (k == 1) {
        return head.next;   // delete head
    }

    int cnt = 0;
    Node temp = head;
    Node prev = null;

    while (temp != null) {
        cnt++;

        if (cnt == k) {
            prev.next = prev.next.next;
            break;
        }

        prev = temp;
        temp = temp.next;
    }

    return head;
}

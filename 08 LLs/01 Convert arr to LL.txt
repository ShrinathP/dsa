/*
1. Create a head node  = first array element
2. Current = head
3. For loop from 1 to arr.length, 
  temp = arr[i],      --(create new temp node)
  curr.next = temp,   --(attach curr's next to temp)
  curr = temp         --(reassign curr to temp)
*/

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

public class ArrayToLinkedList {

    public static Node convertToLL(int[] arr) {
        if (arr == null || arr.length == 0) {
            return null;
        }

        // Step 1: Create head node
        Node head = new Node(arr[0]);
        Node current = head;

        // Step 2: Loop through remaining elements
        for (int i = 1; i < arr.length; i++) {
            Node temp = new Node(arr[i]);  // <-- as requested
            current.next = temp;
            current = temp;
        }

        return head;
    }

    // Helper method to print Linked List
    public static void printLL(Node head) {
        Node temp = head;
        while (temp != null) {
            System.out.print(temp.data + " -> ");
            temp = temp.next;
        }
        System.out.println("null");
    }

    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50};
        Node head = convertToLL(arr);
        printLL(head);
    }
}

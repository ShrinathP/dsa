//Coding Decoded
//Mathematical approach
//C = a*a + b*b
//So find b = Math.sqrt(c - a*a)
//So keep finding b which has no decimals
//find an integer b

public class Solution {
    public boolean judgeSquareSum(int c) {
        for (long a = 0; a * a <= c; a++) 
        {
            double b = Math.sqrt(c - a * a);
            if (b == (int) b)
                return true;
        }
        return false;
    }
}

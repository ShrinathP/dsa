//chatgpt, gemini solution
//2 pointer
/*
1. sort the array
2. i=0,j=1 2 pointers
3. while j<length, if diff is less move j ahead, if diff is more move i ahead
4. else diff match, count++, move both ahead
5. skip duplicates

*/

class Solution {
    public int findPairs(int[] nums, int k) {
        if (k < 0) return 0;

        Arrays.sort(nums);
        int i = 0, j = 1, count = 0;

        while (j < nums.length) {
            if (i == j || nums[j] - nums[i] < k) {
                j++;
            } else if (nums[j] - nums[i] > k) {
                i++;
            } else {
                count++;
                i++;
                j++;

                // skip duplicates
                while (j < nums.length && nums[j] == nums[j - 1]) j++;
            }
        }
        return count;
    }
}

import java.util.Arrays;

//chatgpt solution
//Right to left traversal
//Fix the largest side on the right side = k
//Use two pointers: i, j
//if condition satisfies

/*

✅ Algorithm Steps
1. Sort the array
2. Initialize count = 0
3. Fix the largest side at index k from right to left
4. Use two pointers:
* i = 0 (smallest)
* j = k - 1
5. While i < j:
* If nums[i] + nums[j] > nums[k]
    → all pairs from i to j-1 work
    → count += (j - i)
    → j--
* Else
    → i++
6. Return count
*/
class Solution {
    public int triangleNumber(int[] nums) {
        int n = nums.length;
        if (n < 3) return 0;

        Arrays.sort(nums);
        int count = 0;

        // Fix the largest side
        for (int k = n - 1; k >= 2; k--) {
            int i = 0;
            int j = k - 1;

            while (i < j) {
                if (nums[i] + nums[j] > nums[k]) {
                    // All pairs from i to j-1 are valid
                    count += (j - i);
                    j--;
                } else {
                    i++;
                }
            }
        }

        return count;
    }
}

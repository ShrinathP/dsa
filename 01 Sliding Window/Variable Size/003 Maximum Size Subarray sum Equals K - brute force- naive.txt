/*
Problem description:
Given an array arr[] of size n containing integers, the task is to find the length of the longest subarray having sum equal to the given value k.
*/

class Solution {

    
    //Time Complexity O(N square)
//2 for loops
//Naresh Gupta
class Solution {
    public int maxProduct(int[] nums, int k) {
        int n = nums.length;
        
        int result = 0;
        for(int i=0; i< n; i++){
            int sum = 0;
            for(int j=i; j<n; j++){
                sum += nums[j];
		if(sum == k) {
		result = Math.max(result, j-i + 1);
		}
                
            }
        }
        return result;
        
    }
}

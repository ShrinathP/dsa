/*
Problem
Given a circular array arr[] of size n with distinct elements, 
find the next greater element for each element in the array.

Striver Explanation, gfg soln
https://www.geeksforgeeks.org/dsa/find-the-next-greater-element-in-a-circular-array/
Circular Array - modulo way

Naive Solution, 
for in for
( i + j )%n - virtually doubling array
*/


import java.util.ArrayList;

class GfG {

    static ArrayList<Integer> nextGreater(int[] arr) {
        int n = arr.length;
        int[] res = new int[n];

        for (int i = 0; i < n; i++) {
            res[i] = -1;
        }

        // Iterate for all the elements of the array
        for (int i = 0; i < n; i++) {
            for (int j = 1; j < n; j++) {
                // Checking for next greater element
                if (arr[i] < arr[(i + j) % n]) {
                    res[i] = arr[(i + j) % n];
                    break;
                }
            }
        }

        ArrayList<Integer> result = new ArrayList<>();
        for (int val : res) {
            result.add(val);
        }

        return result;
    }

    public static void main(String[] args) {
        int[] arr = {1, 3, 2, 4};
        ArrayList<Integer> ans = nextGreater(arr);

        for (int i : ans) {
            System.out.print(i + " ");
        }
    }
}

import java.util.*;

public class RemoveKDigits {
    public static String removeKdigits(String num, int k) {
        // Use a stack to build the result
        Stack<Character> stack = new Stack<>();
        
        for (char digit : num.toCharArray()) {
            // Remove digits from the stack if they are greater than the current digit
            // and we still have k digits to remove
            while (!stack.isEmpty() && k > 0 && stack.peek() > digit) {
                stack.pop(); // Pop the larger digit
                k--;
            }
            stack.push(digit);
        }

        // If k > 0, remove remaining digits from the end
        while (k > 0 && !stack.isEmpty()) {
            stack.pop();
            k--;
        }

        // Build the final number
        StringBuilder sb = new StringBuilder();
        for (char c : stack) {
            sb.append(c);
        }

        // Remove leading zeros
        while (sb.length() > 0 && sb.charAt(0) == '0') {
            sb.deleteCharAt(0);
        }

        // If empty, return "0"
        return sb.length() == 0 ? "0" : sb.toString();
    }

    // Test
    public static void main(String[] args) {
        String num = "1432219";
        int k = 3;
        System.out.println(removeKdigits(num, k)); // Output: "1219"

        System.out.println(removeKdigits("10200", 1)); // Output: "200"
        System.out.println(removeKdigits("10", 2));    // Output: "0"
    }
}

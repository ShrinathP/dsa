class Solution {
    public int lengthOfLIS(int[] nums) {
        if (nums.length == 0) return 0;

        int[] tails = new int[nums.length];
        int size = 0;

        for (int num : nums) {
            int idx = lowerBound(tails, 0, size - 1, num);
            tails[idx] = num;
            if (idx == size) {
                size++;
            }
        }
        return size;
    }

    // lower_bound: first index where arr[idx] >= target
    private int lowerBound(int[] arr, int left, int right, int target) {
        int ans = right + 1;   // default insert position

        while (left <= right) {
            int mid = left + (right - left) / 2;

            if (arr[mid] >= target) {
                ans = mid;
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return ans;
    }
}

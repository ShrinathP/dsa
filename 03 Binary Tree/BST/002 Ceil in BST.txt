/*
Striver explanation - 
//if curr val is greater, keep as answer, move left

✅ Explanation:

1. Ceil in a BST = smallest node value ≥ key.
2. Traverse BST:
    If key == node.val → perfect match → return node.val.
    If node.val > key → keep ans, go left to search for better ans.
    Else → move right


*/

public class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode() {}
    TreeNode(int val) { this.val = val; }
    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

public class Solution {

    public static int findCeil(TreeNode<Integer> root, int key) {

        int ceil = -1;
        while (root != null) {
            if (root.data == key) {
                ceil = root.data;
                return ceil;
            }

//if curr val is smaller, keep as answer, move left
            if (root.data > key ) {
                ceil = root.data;
                root = root.left;
            } else {
                root = root.right;
            }
        }
        return ceil;
    }
}


/*
Inorder predecessor

1. Initialize: predecessor = null
2. While root ≠ null:
   a. If root.val < p.val:
       → root may be predecessor → store it
       → move to root.right
   b. Else:
       → move to root.left
3. Return predecessor

*/

class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int x) { val = x; }
}


public TreeNode inorderPredecessor(TreeNode root, TreeNode p) {
    TreeNode predecessor = null;

    while (root != null) {
        if (root.val < p.val) {
//root val is less than p, keep ans, move right
            predecessor = root;      // potential predecessor
            root = root.right;
        } else {
            root = root.left;
        }
    }

    return predecessor;
}

class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int x) { val = x; }
}


public TreeNode inorderSuccessor(TreeNode root, TreeNode p) {
    TreeNode successor = null;

    while (root != null) {
        if (p.val < root.val) {
            successor = root;         // potential successor
            root = root.left;
        } else {
            root = root.right;
        }
    }

    return successor;
}

/*
- Full imaginary Binary tree width- 
The width of a level is measured using positions from a conceptual complete binary tree, so null gaps between nodes are counted. This is why we assign indices to nodes and compute width as rightIndex âˆ’ leftIndex + 1.

chatgpt solution
*/

public class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode() {}
    TreeNode(int val) { this.val = val; }
    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

class Solution {
    public int widthOfBinaryTree(TreeNode root) {
        if (root == null) return 0;

        int maxWidth = 0;
        Queue<Pair<TreeNode, Integer>> q = new LinkedList<>();
        q.offer(new Pair<>(root, 0));

        while (!q.isEmpty()) {
            int size = q.size();
            int minIndex = q.peek().getValue(); // leftmost index at this level
            int first = 0, last = 0;

            for (int i = 0; i < size; i++) {
                Pair<TreeNode, Integer> p = q.poll();
                TreeNode node = p.getKey();
                int idx = p.getValue() - minIndex; // normalize index

                if (i == 0) first = idx;
                if (i == size - 1) last = idx;

                if (node.left != null)
                    q.offer(new Pair<>(node.left, 2 * idx + 1));
                if (node.right != null)
                    q.offer(new Pair<>(node.right, 2 * idx + 2));
            }

            maxWidth = Math.max(maxWidth, last - first + 1);
        }

        return maxWidth;
    }
}

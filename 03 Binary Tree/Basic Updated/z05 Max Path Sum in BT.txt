//gemini solution, striver explanation
class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode() {}
    TreeNode(int val) { this.val = val; }
    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

public class BinaryTreeMaxPathSum {
    // Global variable to store the maximum path sum found so far
    private int maxSum = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        dfs(root);
        return maxSum;
    }

    // DFS helper function that returns the maximum single-branch path sum ending at the current node
    private int dfs(TreeNode node) {
        if (node == null) {
            return 0;
        }

        // Recursively get the maximum path sums from the left and right children
        // We take max(0, ...) to ignore negative contributions from a subtree
        int leftSum = Math.max(0, dfs(node.left));
        int rightSum = Math.max(0, dfs(node.right));

        // Calculate the maximum path sum IF the current node is the 'highest point' (turning point)
        // This path includes the node, its left branch, and its right branch
        int currentPathSum = node.val + leftSum + rightSum;
        // Update the global maximum path sum
        maxSum = Math.max(maxSum, currentPathSum);

        // Return the maximum single-branch path sum that can be extended to the parent
        // This is the node's value plus the larger of its two children's contributions
        return node.val + Math.max(leftSum, rightSum);
    }
}

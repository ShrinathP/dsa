//gemini solution using Pair, striver explanation,
//For each node Pair holds node, hd (horizontal distance)


// Definition for a binary tree node
class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int val) {
        this.val = val;
        this.left = this.right = null;
    }
}

// Class to store the node and its horizontal distance
class Pair {
    TreeNode node;
    int hd; // Horizontal Distance
    Pair(TreeNode node, int hd) {
        this.node = node;
        this.hd = hd;
    }
}

public class BottomViewTree {

    public List<Integer> bottomView(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) {
            return result;
        }

        // TreeMap to store the bottom-most node's value for each horizontal distance.
        // TreeMap keeps the entries sorted by key (HD).
        Map<Integer, Integer> map = new TreeMap<>();
        // Queue for level order traversal, storing Pair objects (node and its HD)
        Queue<Pair> q = new LinkedList<>();

        q.add(new Pair(root, 0));

        while (!q.isEmpty()) {
            Pair current = q.poll();
            TreeNode node = current.node;
            int hd = current.hd;

            // Update the map for the current horizontal distance.
            // In BFS, the last node processed at a specific HD is the bottom-most one.
            map.put(hd, node.val);

            // Enqueue left child with HD - 1
            if (node.left != null) {
                q.add(new Pair(node.left, hd - 1));
            }

            // Enqueue right child with HD + 1
            if (node.right != null) {
                q.add(new Pair(node.right, hd + 1));
            }
        }

        // Extract the node values from the map in sorted order of HD
        for (int value : map.values()) {
            result.add(value);
        }

        return result;
    }

    // Example Usage:
    public static void main(String[] args) {
        BottomViewTree tree = new BottomViewTree();
        // Construct a sample tree:
        //       20
        //      /  \
        //     8    22
        //    / \   / \
        //   5  10 3  25
        //     /  \
        //    ?   14
        // (Note: Node 3 should be at HD 0 in the diagram in search result 0.1.1. 
        // For simplicity, let's use a standard one).
        
        TreeNode root = new TreeNode(1);
        root.left = new TreeNode(2);
        root.right = new TreeNode(3);
        root.left.left = new TreeNode(4);
        root.left.right = new TreeNode(5);
        root.right.left = new TreeNode(6);
        root.right.right = new TreeNode(7);
        root.right.left.right = new TreeNode(8);

        // Sample tree structure from search result 0.1.7 explanation (slightly adjusted):
        //       1
        //      / \
        //     2   3
        //        / \
        //       5   6
        //      / \
        //     7   8
        // Bottom view should be [7, 5, 8, 6] (based on the sample in the snippets)
        
        List<Integer> bottomViewNodes = tree.bottomView(root);
        System.out.println("Bottom View: " + bottomViewNodes);
    }
}

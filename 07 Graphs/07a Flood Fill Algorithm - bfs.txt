// Java implementation of Flood Fill Algorithm
// using BFS

/*
BFS way using Queue 
- add the first element in the queue
- Check every direction of the element - 
- if the direction bound satisfies and color is oldColor - 
- Change the colour and add it to the queue


*/

import java.util.*;

class GfG {

    static class Pair {
        int row;
        int col;

        Pair(int row, int col) {
            this.row = row;
            this.col = col;
        }
    }

    // Up, Right, Down, Left
    

    public static int[][] floodFill(int[][] image, int sr, int sc, int newColor) {

        if (image[sr][sc] == newColor)
            return image;

        int oldColor = image[sr][sc];

        Queue<Pair> q = new LinkedList<>();
        q.offer(new Pair(sr, sc));

        image[sr][sc] = newColor;

        //define the directions before !q.isEmpty()
        int[] delRow = {-1, 0, 1, 0};
        int[] delCol = {0, 1, 0, -1};

        while (!q.isEmpty()) {
            Pair curr = q.poll();
            int row = curr.row;
            int col = curr.col;

            for (int i = 0; i < 4; i++) {

                int nRow = row + delRow[i];
                int nCol = col + delCol[i];

                if (nRow >= 0 && nRow < image.length &&
                    nCol >= 0 && nCol < image[0].length &&
                    image[nRow][nCol] == oldColor) {

                    image[nRow][nCol] = newColor;
                    q.offer(new Pair(nRow, nCol));
                }
            }
        }
        return image;
    }

    public static void main(String[] args) {

        int[][] image = {
            {1, 1, 1, 0},
            {0, 1, 1, 1},
            {1, 0, 1, 1}
        };

        int sr = 1;
        int sc = 2;
        int newColor = 2;

        int[][] result = floodFill(image, sr, sc, newColor);

        for (int[] row : result) {
            for (int pixel : row) {
                System.out.print(pixel + " ");
            }
            System.out.println();
        }
    }
}

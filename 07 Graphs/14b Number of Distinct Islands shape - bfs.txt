//BFS Solution - chatgpt solution
// Add the current row, col element to the queue, set visited to 1
// while queue is not empty, poll from the queue, subtract base coords, add to the arrayList
//move all 4 directions, keep adding the the queue and set visited to 1

import java.util.*;

public class Solution {
    // Directions for 4 possible moves (Up, Down, Left, Right)
    static int[] delRow = {-1, 1, 0, 0};
    static int[] delCol = {0, 0, -1, 1};

    // Helper function for BFS
    private void bfs(int row, int col, int[][] vis, int[][] grid,
                     ArrayList<String> vec, int row0, int col0) {

        int m = grid.length;
        int n = grid[0].length;

        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{row, col});
        vis[row][col] = 1;

        while (!queue.isEmpty()) {
            int[] current = queue.poll();
            int r = current[0];
            int c = current[1];

            // Store relative coordinates
            vec.add(toString(r - row0, c - col0));

            // Visit all 4 neighbours
            for (int i = 0; i < 4; i++) {
                int nRow = r + delRow[i];
                int nCol = c + delCol[i];

                if (nRow >= 0 && nRow < m &&
                    nCol >= 0 && nCol < n &&
                    grid[nRow][nCol] == 1 &&
                    vis[nRow][nCol] == 0) {

                    queue.offer(new int[]{nRow, nCol});
                    vis[nRow][nCol] = 1;
                }
            }
        }
    }

    // Converts relative coordinates to a string
    private String toString(int r, int c) {
        return r + "-" + c;
    }

    // Main function
    public int countDistinctIslands(int[][] grid) {

        int m = grid.length;
        int n = grid[0].length;

        int[][] vis = new int[m][n];

        // Stores each island's shape
        HashSet<ArrayList<String>> st = new HashSet<>();

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {

                if (grid[i][j] == 1 && vis[i][j] == 0) {

                    ArrayList<String> vec = new ArrayList<>();

                    bfs(i, j, vis, grid, vec, i, j);

                    st.add(vec);
                }
            }
        }

        return st.size();
    }
}

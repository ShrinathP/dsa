import java.util.*;

public class MaxOfMinWindow {

    public int[] maxOfMin(int[] arr) {
        int n = arr.length;
        
        int[] left = new int[n];
        int[] right = new int[n];
        
        Stack<Integer> stack = new Stack<>();
        
        // 1. Previous Smaller Element (PSE)
        for (int i = 0; i < n; i++) {
            while (!stack.isEmpty() && arr[stack.peek()] >= arr[i]) {
                stack.pop();
            }
            left[i] = stack.isEmpty() ? -1 : stack.peek();
            stack.push(i);
        }
        
        stack.clear();
        
        // 2. Next Smaller Element (NSE)
        for (int i = n - 1; i >= 0; i--) {
            while (!stack.isEmpty() && arr[stack.peek()] >= arr[i]) {
                stack.pop();
            }
            right[i] = stack.isEmpty() ? n : stack.peek();
            stack.push(i);
        }
        
        // 3. Answer array
        int[] maxMin = new int[n + 1];
        Arrays.fill(maxMin, Integer.MIN_VALUE);
        
        // 4. Fill using window length
        for (int i = 0; i < n; i++) {
            int len = right[i] - left[i] - 1;
            maxMin[len] = Math.max(maxMin[len], arr[i]);
        }

    }
}

public int CelebrityProblem() {
    int top = 0;
    int down = n-1;

    while(top < down) {
        //top knows down, top cannot be celebrity
        if(matrix[top][down] == 1) {
            top = top + 1;
        }

        //down knows top, down cannot be celebrity
        else if(matrix[down][top] == 1) {
            down = down - 1;
        }

        else {
            //both are zero, top doesnt know down and down doesnt know top
            //neither of those can be celebrity, someone has to know the other one
            top--;
            down--;
        }
    }

    //top crosses down, no celebrity
    if(top > down) return -1;

            continue;

            //every element in the column of the top index should be 1, everyone should know top

            // this top condition is still valid

            return -1;
        }

/*
🧩 Problem Summary:
Given n intervals, remove the minimum number of intervals to make the rest non-overlapping.

Google Gemini answer
more easier way to understand
*/


import java.util.Arrays;
import java.util.Comparator;

public class NonOverlappingIntervals {

    /**
     * Finds the minimum number of intervals to remove to make the rest non-overlapping.
     * Intervals that only touch at a point (e.g., [1, 2] and [2, 3]) are considered non-overlapping.
     *
     * @param intervals a 2D array where each intervals[i] = [start_i, end_i]
     * @return the minimum number of removals
     */
    public int eraseOverlapIntervals(int[][] intervals) {
        if (intervals.length == 0) {
            return 0;
        }

        // 1. Sort intervals by their end times in ascending order.
        // Sorting by end time is crucial for the greedy strategy to work.
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]);

        int countKept = 1; // Start with the first interval, which is always kept after sorting
        int currentEnd = intervals[0][1]; // Track the end time of the last kept interval

        // 2. Iterate through the sorted intervals starting from the second one.
        for (int i = 1; i < intervals.length; i++) {
            int nextStart = intervals[i][0];
            int nextEnd = intervals[i][1];

            // 3. If the current interval's start time is >= the last kept interval's end time,
            // they do not overlap, so we can keep the current interval.
            if (nextStart >= currentEnd) {
                countKept++;
                currentEnd = nextEnd; // Update the end time to the current interval's end
            }
            // If they overlap (nextStart < currentEnd), we skip the current interval (effectively "removing" it)
            // because keeping the one with the earlier end time (which is already 'currentEnd')
            // leaves more room for subsequent intervals.
        }

        // The minimum number of removals is the total intervals minus the maximum number we could keep.
        return intervals.length - countKept;
    }

    // Example Usage:
    public static void main(String[] args) {
        NonOverlappingIntervals solution = new NonOverlappingIntervals();
        int[][] intervals1 = {{1, 2}, {2, 3}, {3, 4}, {1, 3}};
        System.out.println("Removals for intervals1: " + solution.eraseOverlapIntervals(intervals1)); // Output: 1

        int[][] intervals2 = {{1, 2}, {1, 2}, {1, 2}};
        System.out.println("Removals for intervals2: " + solution.eraseOverlapIntervals(intervals2)); // Output: 2

        int[][] intervals3 = {{1, 2}, {2, 3}};
        System.out.println("Removals for intervals3: " + solution.eraseOverlapIntervals(intervals3)); // Output: 0
    }
}

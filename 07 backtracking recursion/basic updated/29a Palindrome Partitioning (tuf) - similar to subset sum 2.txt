 /*
 Problem Description
 Given a string s, the task is to find the minimum number of cuts needed for palindrome partitioning of the given string.
 A partitioning of the string is a palindrome partitioning if every sub-string of the partition is a palindrome.

Example
    Input: s = "geek" 
    Output: 2 
    Explanation: We need to make minimum 2 cuts, i.e., "g | ee | k".

 Striver Explanation
 IMP Cue - recurMethod
         -> reach s.length,add to result
         -> for loop(isPalindrome, add substring, recurMethod(index+1), remove)
=============================================================================
Steps
  1. Init, result ArrayList of List, currPath  List, isPalindrome(string, start, end) utility function
  2. recurMethod method(index, string, currPath, result)
    2.1 - Return Statement - we reached end
    if(index = string.length()), add result.add(currPath), return
    2.2 - For loop (i=index;i<s.length;i++)
        - If s[index..i] is a palindrome, isPalindrome(s, index, i)
            - add to currPath -> currPath.add(s.substring(index, i+1))
            - recurMethod with index + 1
            - remove last from currPath -> currPath.remove(currPath.size()-1)
=============================================================================
 */


    //Take u forward solution
    //Pure recursion

import java.util.*;

class Solution {
    

    // Main function to return all palindrome partitions
    public List<List<String>> partition(String s) {
        // Initialize result and temporary path
        List<List<String>> res = new ArrayList<>();
        List<String> path = new ArrayList<>();
        // Start backtracking from index 0
        backtrack(0, s, path, res);
        return res;
    }

    // Backtracking function to generate all possible palindrome partitions
    private void backtrack(int index, String s, List<String> path, List<List<String>> res) {
        // If index reached the end of string, add the current path to result
        if (index == s.length()) {
            res.add(new ArrayList<>(path));
            return;
        }

        // Try all possible substrings starting from index
        for (int i = index; i < s.length(); i++) {
            // Check if current substring is palindrome
            if (isPalindrome(s, index, i)) {
                // Add substring to current path
                path.add(s.substring(index, i + 1));
                // Recur for the next index
                backtrack(i + 1, s, path, res);
                // Backtrack by removing the last added substring
                path.remove(path.size() - 1);
            }
        }
    }

    // Function to check if a given substring is a palindrome
    private boolean isPalindrome(String s, int start, int end) {
        // Loop while start is less than end
        while (start < end) {
            // If characters at start and end are not equal, it's not a palindrome
            if (s.charAt(start) != s.charAt(end)) return false;
            // Move start forward and end backward
            start++;
            end--;
        }
        // If all characters matched, it's a palindrome
        return true;
    }

}

// Driver class to test the solution
public class Main {
    public static void main(String[] args) {
        // Input string
        String s = "aab";
        // Create object of Solution
        Solution sol = new Solution();
        // Get all partitions
        List<List<String>> result = sol.partition(s);

        // Print the partitions
        for (List<String> part : result) {
            for (String str : part) {
                System.out.print(str + " ");
            }
            System.out.println();
        }
    }
}

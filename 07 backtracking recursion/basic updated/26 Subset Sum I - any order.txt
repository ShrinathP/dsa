/*
Given an array arr[], print all subset sums (not combinations) in any order.
Chatgpt soln
*/



import java.util.*;

class Solution {
    public List<Integer> subsetSums(int[] arr) {
        List<Integer> result = new ArrayList<>();
        helper(arr, 0, 0, result);
        return result;
    }

    private void helper(int[] arr, int index, int sum, List<Integer> result) {
        if (index == arr.length) {
            result.add(sum);
            return;
        }

        // Include current element - increase the sum
        helper(arr, index + 1, sum + arr[index], result);

        // Exclude current element - dont increase the sum
        helper(arr, index + 1, sum, result);
    }

    // Example usage
    public static void main(String[] args) {
        Solution sol = new Solution();
        int[] arr = {1, 2, 3};
        System.out.println(sol.subsetSums(arr)); // Output: [6, 3, 4, 0, 5, 2, 3, 0] (any order)
    }
}

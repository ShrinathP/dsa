import java.util.*;

class Solution {
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(candidates); // Sort to handle duplicates
        backtrack(candidates, 0, target, new ArrayList<>(), result);
        return result;
    }

    private void backtrack(int[] cand, int start, int target, List<Integer> temp, List<List<Integer>> result) {
        if (target == 0) {
            result.add(new ArrayList<>(temp));
            return;
        }

        for (int i = start; i < cand.length; i++) {
            
            if (i > start && cand[i] == cand[i - 1]) continue; // skip duplicates no set needed
            // Ensure to skip duplicates at the same recursive level, not across levels.
            
            if (cand[i] > target) break; // prune

            temp.add(cand[i]);
            backtrack(cand, i + 1, target - cand[i], temp, result);
            temp.remove(temp.size() - 1); // backtrack
        }
    }
}

public class CountSquares {

    public static int countSquares(int[][] arr) {
        int n = arr.length;
        int m = arr[0].length;

        // dp[i][j] = size of largest square ending at (i, j)
        int[][] dp = new int[n][m];

        // ---------------------------
        // Step 1: initialize first row
        // Only 1x1 squares possible here
        // ---------------------------
        for (int j = 0; j < m; j++) {
            dp[0][j] = arr[0][j];
        }

        // ---------------------------
        // Step 2: initialize first column
        // Only 1x1 squares possible here
        // ---------------------------
        for (int i = 0; i < n; i++) {
            dp[i][0] = arr[i][0];
        }

        // ---------------------------
        // Step 3: fill DP table
        // For each cell, if it is 1,
        // we try to expand a square using:
        // top, left, and top-left neighbors
        // ---------------------------
        for (int i = 1; i < n; i++) {
            for (int j = 1; j < m; j++) {

                // If cell is 0, no square can end here
                if (arr[i][j] == 0) {
                    dp[i][j] = 0;
                } else {
                    // Extend the minimum square from neighbors
                    dp[i][j] = 1 + Math.min(
                        dp[i - 1][j],          // top
                        Math.min(dp[i - 1][j - 1], dp[i][j - 1]) // diagonal & left
                    );
                }
            }
        }

        // ---------------------------
        // Step 4: sum all dp values
        // Each dp[i][j] contributes:
        // number of squares ending at (i, j)
        // ---------------------------
        int sum = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum += dp[i][j];
            }
        }

        return sum;
    }

    public static void main(String[] args) {
        int[][] matrix = {
            {0,1,1,1},
            {1,1,1,1},
            {0,1,1,1}
        };

        System.out.println(countSquares(matrix)); // Output: 15
    }
}

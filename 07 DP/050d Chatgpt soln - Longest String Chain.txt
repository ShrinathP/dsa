/*
Problem Statement
🧩 Longest String Chain — Problem
Given a list of words, return the length of the longest possible chain:
A word A is a predecessor of B if you can insert exactly one character anywhere in A to get B.

💡 Key Idea (Sort + LIS-style DP)
Sort words by length
For each word i, check all smaller words j
If j is a valid predecessor of i → extend chain
*/



import java.util.*;

public class Solution {

    public int longestStrChain(String[] words) {
        int n = words.length;
        
        Arrays.sort(words, (a, b) -> a.length() - b.length());
        
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        
        int maxLen = 1;
        
        for (int i = 0; i < n; i++) {
            for (int prev = 0; prev < i; prev++) {
                if (isPredecessor(words[prev], words[i]) && dp[prev] + 1 > dp[i]) {
                    dp[i] = dp[prev] + 1;
                }
            }
            maxLen = Math.max(maxLen, dp[i]);
        }
        
        return maxLen;
    }

    // check if s1 -> s2 (s1 smaller)
    private boolean isPredecessor(String s1, String s2) {
      //length check s2 should be s1 + 1
        if (s2.length() != s1.length() + 1) return false;
        
        //init i,j=0, 
        //while j < s2.length, compare if match increase i, increase j in any case, if i reaches s1 end, return true
        int i = 0, j = 0;
        
        while (j < s2.length()) {
            if (i < s1.length() && s1.charAt(i) == s2.charAt(j)) {
                i++;
            }
            j++;
        }
        
        return i == s1.length();
    }
}


/*
Extension of Previous Problem
Given an array arr[] and a number d, count the number of ways to split the array into two subsets S1 and S2 such that:
S1 - S2 = d

IMP CUE -> This is Previous Problem With modified target 
--------------------------------------------------------

Reduction to Subset Sum 
Equation1
S1 - S2 = D

Equation2
S1 + S2 = totalSum
S1 = totalSum - S2

Equation1 will become
(totalSum - S2) - S2 = D
totalSum - D = 2S2
S2 = (totalSum - D)/2

Similarly
S1 = (totalSum + D)/2

So we have to find subsets with sum either (totalSum - D)/2 or (totalSum + D)/2

2 Base cases -
Base Case 1 -> totalSum has to be greater than d, totalSum > d, if less return false
Base Case 2 -> totalSum - D has to be even, if off then subset cant be split - as we dont have decimals

*/

// Space Optimized 1D Array Solution 

class Solution {
    public int countPartitions(int n, int d, int[] arr) {

        int totalSum = 0;
        for (int num : arr) totalSum += num;

        // Base Case 1 and 2
        if (totalSum < d || (totalSum + d) % 2 != 0) return 0;

        int target = (totalSum + d) / 2;
        //int target = (totalSum - d) / 2;

        return countSubsets(arr, target);
    }

// Previous Space Optimized Solution
    private int countSubsets(int[] arr, int target) {
        int n = arr.length;
        int[] prev = new int[target + 1];

        // Base case
        prev[0] = 1;

        // Handle first element explicitly (important for zeros)
        if (arr[0] == 0) {
            prev[0] = 2;  // pick or not pick
        } else if (arr[0] <= target) {
            prev[arr[0]] = 1;
        }

        for (int i = 1; i < n; i++) {

            int[] curr = new int[target + 1];

            // sum = 0 always possible
            curr[0] = 1;

            for (int sum = 0; sum <= target; sum++) {

                int notTake = prev[sum];

                int take = 0;
                if (arr[i] <= sum) {
                    take = prev[sum - arr[i]];
                }

                curr[sum] = take + notTake;
            }

            prev = curr;
        }

        return prev[target];
    }
}

/*
Longest Common String from S1 and S2
Substring -> contiguous Subsequence
Check Striver explanation
Same as LCS Tabulation solution but remove the not match case 
fill dp[i][j] = 0 instead of removing either of the chars Math.max(dp[i-1][j], dp[i][j-1])
this is a special kind of subsequence  we cant skip not matches, for not match we fill dp = 0
*/



class Solution {
    public int longestCommonSubstring(String s1, String s2) {
        int n = s1.length(), m = s2.length();

        int[][] dp = new int[n + 1][m + 1];
        int maxLen = 0;

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {

                if (s1.charAt(i - 1) == s2.charAt(j - 1)) {
                    dp[i][j] = 1 + dp[i - 1][j - 1];
                    maxLen = Math.max(maxLen, dp[i][j]);
                } else {
                    dp[i][j] = 0; // 🔥 reset instead of remove either and search
                }

            }
        }

        return maxLen;
    }
}

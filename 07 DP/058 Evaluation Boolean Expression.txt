/*
Problem: Evaluate Boolean Expression to True

You are given:
A string exp consisting of:
'T' (true), 'F' (false)
Operators: & (AND), | (OR), ^ (XOR)

Task:
Count the number of ways to parenthesize the expression such that it evaluates to True


Example String
T ^ F | T & F | T
i -> index

At every operand we can create a partition
Operands start at i index and are at a distance of i+2 indexes

consider arr splitted to arr2 and arr2
If the operand i.e arr[i] is 
For AND
& -> For result to be TRUE  -> arr1 and arr2 both have to be true, leftTrue and rightTrue
  -> For result to be FALSE -> arr1 true and arr2 false, arr1 false and arr2 true, arr1 false and arr2 false

For OR
| -> For result to be TRUE  -> arr1 true and arr2 false, arr1 false and arr2 true, arr1 true and arr2 true
  -> For result to be FALSE -> arr1 false and arr2 false

For XOR
^ -> For result to be TRUE  -> arr1 true and arr2 false, arr1 false and arr2 true
  -> For result to be FALSE -> arr1 false and arr2 false, arr1 true and arr2 true

==========================================================================================
Steps
1. recurFunction(expression, int i, int j, isTrue)
    recurFunction(String, 0, n-1, 1) //traverse from 0 to n-1
  1.1 Base case, if i > j then return 0, no substring present, return 0 ways
  1.2 if i==j, single character, if isTrue expected, check for exp.charAt(i) for 'T', if false then check for 'F'
      if(i==j) {
        if(isTrue) return exp.charAt(i) == 'T' ? 1: 0;
        return exp.charAt(i) == 'F' ? 1: 0;
      }
  1.3 Modify partitionIndex to pick only operands
      For all the partitionIndex from i+1 to j-1, with +2 jumps
      - recur and calc leftTrue, leftFalse, rightTrue, rightFalse
      - get kth char operand
      - check kth operand for & ^ |, and add ways based on above logic
  1.4 return ways
==========================================================================================

*/



public class BooleanExpression {

    public static int solve(String exp, int i, int j, boolean isTrue) {

        // 🔹 Base case
        if (i > j) return 0;

        // single character
        if (i == j) {
            if (isTrue) return exp.charAt(i) == 'T' ? 1 : 0;
            else return exp.charAt(i) == 'F' ? 1 : 0;
        }

        int ways = 0;

        // partition at operators
        for (int k = i + 1; k <= j - 1; k += 2) {

            char op = exp.charAt(k);

            int lt = solve(exp, i, k - 1, true);
            int lf = solve(exp, i, k - 1, false);
            int rt = solve(exp, k + 1, j, true);
            int rf = solve(exp, k + 1, j, false);

            if (op == '&') {
                if (isTrue) {
                    ways += lt * rt;
                } else {
                    ways += lt * rf + lf * rt + lf * rf;
                }
            } else if (op == '|') {
                if (isTrue) {
                    ways += lt * rt + lt * rf + lf * rt;
                } else {
                    ways += lf * rf;
                }
            } else if (op == '^') {
                if (isTrue) {
                    ways += lt * rf + lf * rt;
                } else {
                    ways += lt * rt + lf * rf;
                }
            }
        }

        return ways;
    }

    public static int countWays(String exp) {
        int n = exp.length();
        return solve(exp, 0, n - 1, true);
    }

    public static void main(String[] args) {
        String exp = "T|F&T^T";
        System.out.println(countWays(exp));
    }
}

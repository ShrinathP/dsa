/*
You cannot Buy right after sell, you have to wait for 1 day
Change from 041a Buy and Sell stock 3 - Recursion - 
is In Sell Step after sell recurMethod with index + 2 instead of index + 1, as we arent allowed to buy on index + 1
*/

class Solution {
    public int maxProfit(int[] prices) {
        return f(0, 1, prices);
    }

    int f(int i, int canBuy, int[] prices) {
        if (i >= prices.length) return 0;

        if (canBuy == 1) {
            int buy = -prices[i] + f(i + 1, 0, prices);
            int skip = f(i + 1, 1, prices);
            return Math.max(buy, skip);
        } else {
            int sell = prices[i] + f(i + 2, 1, prices); // cooldown change
            int skip = f(i + 1, 0, prices);
            return Math.max(sell, skip);
        }
    }
}

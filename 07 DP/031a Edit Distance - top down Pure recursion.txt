/*
Problem Statement : 
Convert s1 → s2 using:
Insert
Delete
Replace
👉 Min operations required

Consider
s1 = horse
s2 = ros

4 cases 
If char matches -> move both i and j back
In insertion - dont move i back, hypothetically we are inserting ahead of i
In deletion - move i back, but j stays the same
In replacement - we will replace with the matching char so move both i and j back

Base cases
if i<0 , s1 exhausted, all pending j need insertions so -> return j + 1
if j<0 , s2 exhausted, all pending i need deletions so -> return i + 1

Steps
1. recurMethod(i, j, s1, s2), i for s1, j for s2
    1.1 Base cases
        if i<0 , s1 exhausted, all pending j need insertions so -> return j + 1
        if j<0 , s2 exhausted, all pending i need deletions so -> return i + 1
    2 cases, if match, if doesnt match
    1.2 If match -> 
        If char match -> 0 operations needed, recurMethod(i-1, j-1, ...) move both back 
        if (s1.charAt(i) == s2.charAt(j)) { return 0 + recurMethod(i-1, j-1, ...)}
    1.3 If not match -> 
        3 other cases, insertion, deletion, replacement
        - insert -> insert ahead of i, i stays the same move j back
        int insert = solve(i, j - 1, s1, s2);
        - delete -> j stays the same, move i back
        int delete = solve(i - 1, j, s1, s2);
        - replace -> move both i and j back
        int replace = solve(i - 1, j - 1, s1, s2);
    1.4 1 operation of above 3, + find min of all
        - return 1 + Math.min(insert, Math.min(delete, replace))
*/

class Solution {
    public int minDistance(String s1, String s2) {
        return solve(s1.length() - 1, s2.length() - 1, s1, s2);
    }

    private int solve(int i, int j, String s1, String s2) {
        if (i < 0) return j + 1; // insert all
        if (j < 0) return i + 1; // delete all

        if (s1.charAt(i) == s2.charAt(j)) {
            return solve(i - 1, j - 1, s1, s2);
        }

        int insert = solve(i, j - 1, s1, s2);
        int delete = solve(i - 1, j, s1, s2);
        int replace = solve(i - 1, j - 1, s1, s2);

        return 1 + Math.min(insert, Math.min(delete, replace));
    }
}

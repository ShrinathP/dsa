Changes from the prev solution
Change 1 -> dp array - to store the minCuts at ith place, fill with -1, pass in the recurFunction
Change 2 -> After base case - check if dp[i] != -1, return dp[i]
Change 3 -> save to dp before return, return dp[i] = minCuts;
*/


import java.util.*;

public class PalindromePartitionMemo {

    // check palindrome
    public static boolean isPalindrome(String s, int i, int j) {
        while (i < j) {
            if (s.charAt(i) != s.charAt(j)) return false;
            i++;
            j--;
        }
        return true;
    }

    public static int solve(String s, int i, int[] dp) {

        // 🔹 Base case
        if (i == s.length()) return 0;

        // 🔹 Memo check
        if (dp[i] != -1) return dp[i];

        int minCuts = Integer.MAX_VALUE;

        for (int j = i; j < s.length(); j++) {

            if (isPalindrome(s, i, j)) {

                int cuts = 1 + solve(s, j + 1, dp);
                minCuts = Math.min(minCuts, cuts);
            }
        }

        return dp[i] = minCuts;
    }

    public static int minCut(String s) {
        int n = s.length();

        int[] dp = new int[n];
        Arrays.fill(dp, -1);

        return solve(s, 0, dp) - 1;
    }

    public static void main(String[] args) {
        String s = "aab";
        System.out.println(minCut(s));
    }
}

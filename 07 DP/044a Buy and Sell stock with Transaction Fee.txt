/*
This is LeetCode 714 – Buy & Sell Stock with Transaction Fee
👉 Same as Buy and Sell stock II, just subtract fee on SELL
Another Variant of Buy and Sell stock 2
*/

class Solution {
    public int maxProfit(int[] prices, int fee) {
        return f(0, 1, prices, fee);
    }

    int f(int i, int canBuy, int[] prices, int fee) {
        if (i == prices.length) return 0;

        if (canBuy == 1) {
            int buy = -prices[i] + f(i + 1, 0, prices, fee);
            int skip = f(i + 1, 1, prices, fee);
            return Math.max(buy, skip);
        } else {
//IMP change, incur fee whenever we sell
            int sell = prices[i] - fee + f(i + 1, 1, prices, fee); 
            int skip = f(i + 1, 0, prices, fee);
            return Math.max(sell, skip);
        }
    }
}

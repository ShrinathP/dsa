IMP -> Maintain a 2d memo array
 int[][] memo = new int[n][target+1] array, can we form a target at index or not ?

So we need 3 states, -1 uncomputed, 0 false, 1 true
boolean [][] can have only 2 states, hence we are using int[][] memo 2d array

/*
Changes from prev Solution
==========================
Change 1 -> Init 2d memo array with arr.length,target + 1 -> int[][] memo = new int[n][target + 1]
(IMP - for target there will be target + 1 states including 0)
Change 2 -> After base cases check memo[index, target]
            if(memo[index][target] != -1) return memo[index][target] == 1;
Change 3 -> before returning take || notTake, save to memo
            memo[index][target] = (take || notTake) ? 1 : 0
            return take || notTake
*/

import java.util.Arrays;

public class SubsetSumMemoization {
    
    public static boolean subsetSum(int[] arr, int target) {
        int n = arr.length;
        // Use an Integer array to store 3 states: null (uncomputed), 1 (true), 0 (false)
        int[][] memo = new int[n][target + 1];
        for (int[] row : memo) Arrays.fill(row, -1);
        
        return solve(n - 1, target, arr, memo);
    }

    private static boolean solve(int ind, int target, int[] arr, int[][] memo) {
        // Base cases
        if (target == 0) return true; // Target reached
        if (ind == 0) return arr[0] == target; // Only one element left
        
        // Return cached result if available
        if (memo[ind][target] != -1) return memo[ind][target] == 1;

        // Choice 1: Exclude the current element
        boolean notTaken = solve(ind - 1, target, arr, memo);

        // Choice 2: Include the current element (if it fits)
        boolean taken = false;
        if (arr[ind] <= target) {
            taken = solve(ind - 1, target - arr[ind], arr, memo);
        }

        // Save result in memo table and return
        memo[ind][target] = (notTaken || taken) ? 1 : 0;
        return notTaken || taken;
    }

    public static void main(String[] args) {
        int[] arr = {1, 2, 3, 4};
        int target = 4;
        if (subsetSum(arr, target)) {
            System.out.println("Subset with the given target found");
        } else {
            System.out.println("Subset with the given target not found");
        }
    }
}

/*
Take not take way, 
While not taking
move index ahead, keep prev same

While taking
keep hold of the prev integer
compare current nums[index] > prev, we can take 1 + recur
move index ahead, consider current nums[index] as prev
*/

class Solution {
    public int lengthOfLIS(int[] nums) {
        return solve(nums, 0, Integer.MIN_VALUE);
    }

    int solve(int[] nums, int index, int prev) {
        if (index == nums.length) return 0;

        // Option 1: skip
        int notTake = solve(nums, index + 1, prev);

        // Option 2: take (only if valid)
        int take = 0;
        if (nums[index] > prev) {
            take = 1 + solve(nums, index + 1, nums[index]);
        }

        return Math.max(take, notTake);
    }
}

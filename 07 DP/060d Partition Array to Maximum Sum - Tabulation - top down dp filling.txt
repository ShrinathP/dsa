/*
Recur moved from front to back
So Tabulation should move from back to front
i.e. 
i will move from n-1 to <=0
j will move from i+1 to Math.min(i+k, n)
*/


import java.util.*;

public class PartitionArrayMaxSumTab {
    public static int maxSumAfterPartitioning(int[] arr, int k) {
        int n = arr.length;
        int[] dp = new int[n + 1]; // dp[i] = max sum from i → end

        dp[n] = 0; // base case: no elements left

        // fill from back → ensures dp[j+1] already computed
        for (int i = n - 1; i >= 0; i--) {
            int maxSum = 0;
            int currMax = 0;

            // try partitions starting at i of size ≤ k
            for (int j = i; j < Math.min(i + k, n); j++) {
                currMax = Math.max(currMax, arr[j]); // track max in current window
                int len = j - i + 1; // current partition length

                // contribution = max * length + best of remaining
                int sum = currMax * len + dp[j + 1];

                maxSum = Math.max(maxSum, sum);
            }

            dp[i] = maxSum; // store best answer for index i
        }

        return dp[0]; // answer from start
    }

    public static void main(String[] args) {
        int[] arr = {1, 15, 7, 9, 2, 5, 10};
        int k = 3;
        System.out.println(maxSumAfterPartitioning(arr, k));
    }
}

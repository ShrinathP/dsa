
/*
Problem Statement
Given an array prices[] representing stock prices, 
find the maximum total profit that can be earned by buying and selling the stock any number of times.

Note: We can only sell a stock which we have bought earlier and we cannot hold multiple stocks on any day.
https://www.geeksforgeeks.org/dsa/stock-buy-sell/

IMP - Greedy approach, as soon as we see profit, sell and buy again
*/


//Kevin naughton
//Keep a look on the next price
//If next price is greater than current price
//we buy today and sell tomorrow, maximizing the profit, if the next price is lesser we hold the stock
//keep adding profits

class Solution {
    public int maxProfit(int[] prices) {
        if(prices == null || prices.length ==0){
            return 0;
        }
        
        int profit = 0;
        for(int i=0; i<prices.length-1; i++){
            if(prices[i+1] > prices[i]){
                profit += prices[i+1] - prices[i];
            }
        }
        
        return profit;
    }
}

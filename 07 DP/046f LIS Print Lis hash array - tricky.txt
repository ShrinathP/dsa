/*
We need to hold the previousIndex where 1 was added
now the print lis solution using another array lets say hash, which holds the previous index where 1 was added
Init hash - fill with -1

IMP: PICKING THE BEST POSSIBLE PREV FOR THE CURRENT I
whenever we check if the condition satisfies i.e. dp[prev] + 1 > dp[i] i.e. we are going to update i.e. update hash[i] to that prev index
as we will pick this prev for the i, so best possible prev for the current i


Striver explanation - Gemini solution
*/

import java.util.*;

public class Solution {
    public static List<Integer> printLIS(int[] nums) {
        if (nums == null || nums.length == 0) {
            return new ArrayList<>();
        }
        
        int n = nums.length;
        int[] dp = new int[n];
        int[] hash = new int[n]; // Tracks the previous index in the LIS sequence
        
        Arrays.fill(dp, 1);
        // Initialize hash: no previous element initially
        Arrays.fill(hash, -1);
        
        int maxLISLength = 1;
        int lastIndex = 0; // Tracks the ending index of the longest subsequence
        
        // 1. Tabulation and parent tracking
        for (int i = 1; i < n; i++) {
            for (int prev = 0; prev < i; prev++) {
                if (nums[i] > nums[prev] && dp[prev] + 1 > dp[i]) {
                    dp[i] = dp[prev] + 1;
                    hash[i] = prev; // Store the index from which we extended the LIS
                }
            }
            // Update the global maximum length and its corresponding ending index
            if (dp[i] > maxLISLength) {
                maxLISLength = dp[i];
                lastIndex = i;
            }
        }
        
        // 2. Backtrack using the hash array to reconstruct the LIS
        List<Integer> lis = new ArrayList<>();
        
        while (lastIndex != -1) {
            lis.add(nums[lastIndex]);
            lastIndex = hash[lastIndex];
        }
        
        // The list is constructed backwards, so reverse it to get correct order
        Collections.reverse(lis);
        return lis;
    }

    public static void main(String[] args) {
        int[] nums = {10, 9, 2, 5, 3, 7, 101, 18};
        List<Integer> result = printLIS(nums);
        System.out.println("Longest Increasing Subsequence: " + result); 
    }
}


/*
Striver explanation - chatgpt solution and explanation
Tricky 
====================================================================================
Steps
1. Init 1d prev array dp with size target + 1
2. Base cases 
    -> prev[0] = 1 - zero target
    -> first Element target arr[0]-> set to 1 - increment by 1
        prev[arr[0]] += 1;
    NOTE IMP:
    prev[arr[0]] += 1; doing this because arr[0] can be 0, so prev[0] will already be 1, 
    so we need to add 1 to it, 2 ways, instead of setting to 1
3. For index(rowindex) in for sum loop, - create new curr arr - fill it
      - init curr row, prefill 0 target =1
      - for sum = 0 to target
          -> do take, not take, save to curr[sum] = take + notTake
          - int notTake = prev[sum]
          - int take = 0;
            if(arr[i] < sum) {take = prev[sum - arr[i]]}
          - curr[sum] = take + notTake
      - assign prev = curr
4. return prev[target]
====================================================================================
      

*/

class Solution {
    public int countSubsets(int[] arr, int n, int target) {

        int[] prev = new int[target + 1];

        // Base case: sum = 0 → always 1 way (empty subset)
        prev[0] = 1;

        // First element handling
        if (arr[0] <= target) {
            prev[arr[0]] += 1;
        }

        for (int i = 1; i < n; i++) { //rowIndex, create new curr arr
            int[] curr = new int[target + 1];

            // sum = 0 always 1 (not taking anything)
            curr[0] = 1;

            for (int sum = 0; sum <= target; sum++) {

                int notTake = prev[sum];

                int take = 0;
                if (arr[i] <= sum) {
                    take = prev[sum - arr[i]];
                }

                curr[sum] = take + notTake;
            }

            // move current row to previous
            prev = curr;
        }

        return prev[target];
    }
}

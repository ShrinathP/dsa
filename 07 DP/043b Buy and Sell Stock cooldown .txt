/*
Not allowed to Buy the next day of Sell
Recursion + Memoization solution - Bottom up Recursion


Changes from previous solution
Change 1. Initialize 2d memo array with size n (days) x 2 (states 0 or 1), fill with -1, pass in recurMethod
Change 2. After base case, if the memo[day][canBuy] != -1, return memo[day][canBuy]
Change 3. before return save the profit to memo[day][canBuy]
*/


import java.util.Arrays;

class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        // dp[i][j]: i is the day (0 to n), j is the state (0 = cannot buy, 1 = can buy)
        int[][] memo = new int[n][2];
        
        // Initialize the memoization table with -1
        for (int[] row : memo) {
            Arrays.fill(row, -1);
        }
        
        // Start from day 0, and you are allowed to BUY initially (state = 1)
        return solve(prices, 0, 1, memo);
    }
    
    private int solve(int[] prices, int day, int canBuy, int[][] memo) {
        // Base Case: If we've processed all days, no more profit can be made
        if (day >= prices.length) {
            return 0;
        }
        
        // Return the cached result if it's already computed
        if (memo[day][canBuy] != -1) {
            return memo[day][canBuy];
        }
        
        int profit = 0;
        
        if (canBuy == 1) {
            // Option 1: Buy the stock today and transition to the "cannot buy" state (canBuy = 0)
            int buyStock = -prices[day] + solve(prices, day + 1, 0, memo);
            // Option 2: Do nothing and keep the "can buy" state (canBuy = 1)
            int skipStock = solve(prices, day + 1, 1, memo);
            
            profit = Math.max(buyStock, skipStock);
        } else {
            // Option 1: Sell the stock today. Due to cooldown, we skip the next day (day + 2)
            // Transition back to "can buy" state (canBuy = 1)
            int sellStock = prices[day] + solve(prices, day + 2, 1, memo);
            // Option 2: Do nothing and keep the "cannot buy" state (canBuy = 0)
            int skipStock = solve(prices, day + 1, 0, memo);
            
            profit = Math.max(sellStock, skipStock);
        }
        
        // Memoize and return the result for the current state
        return memo[day][canBuy] = profit;
    }
}

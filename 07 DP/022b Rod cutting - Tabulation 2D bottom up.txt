

/*
Correct this steps

Bottom up dp filling - similar to 01 Knapsack difference is - infinite supply and the weights value
Steps
1. Init dp arr, n X (n + 1) -> n index , n + 1 lenghts
2. Base case first row - for all lengths from 0 to N(len or n), I can cut 1,1,1 until len is reached, so len * price[0]
// Base case - at index 0, for all lengths price will be = price[0](price at rodLength 1) * len
    for(int cap=0; cap<=capacity; cap++) {dp[0][cap] = (cap / wt[0]) * val[0]; }
3. for in for loop - index 1 to <n i.e. n-1, for len 0 to len(N+1)
    - notTake -> prev row value -> dp[i-1][len]
    - take -> init as INT.MIN, if rodLen <= len take it and reduce len
      take = INT.MIN, if(rodLen <= len) {take = price[i] + dp[i][len - rodLen]}
    - Fill dp
      dp[i][len] = Math.max(take, notTake)
4. return dp[n-1][W]; 
    0 based indexing so row goes from 0 to n-1 and col goes from 0 to W
    0 based indexing so the last row index is n-1 and last col index is W
*/

class Solution {

    public int cutRod(int[] price, int n) {

        int[][] dp = new int[n][n + 1];

        // Base case - at index 0, for all lengths price will be = price[0](price at rodLength 1) * len
        for (int len = 0; len <= n; len++) {
            dp[0][len] = len * price[0];
        }

        for (int i = 1; i < n; i++) {
            for (int len = 0; len <= n; len++) {

                int notTake = dp[i - 1][len];

                int take = Integer.MIN_VALUE;
                int rodLen = i + 1;

                if (rodLen <= len) {
                    take = price[i] + dp[i][len - rodLen]; // same row
                }

                dp[i][len] = Math.max(take, notTake);
            }
        }

        return dp[n - 1][n];
    }
}

Gemini code
Create a dp array with size n = prices.length cross 2 states i.e. 0, 1
Base case - at the end n,0 and n,1 profit is 0
Top down Table filling - opposite of bottom up recursion.
for i-1 to 0, profit will be max at 0th index, 0th state i.e. can buy state

public class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        if (n == 0) return 0;
        
        // dp[i][0] -> Max profit if you can buy stock on day i
        // dp[i][1] -> Max profit if you can sell stock on day i
        int[][] dp = new int[n + 1][2];
        
        // Base case: on day 'n', profit is 0
        dp[n][0] = 0;
        dp[n][1] = 0;
        
        for (int i = n - 1; i >= 0; i--) {
            // Option 1: Buy today (-prices[i] + profit if sold tomorrow), OR Don't buy today (profit if bought tomorrow)
            dp[i][0] = Math.max(-prices[i] + dp[i + 1][1], dp[i + 1][0]);
            
            // Option 2: Sell today (+prices[i] + profit if bought tomorrow), OR Don't sell today (profit if sold tomorrow)
            dp[i][1] = Math.max(prices[i] + dp[i + 1][0], dp[i + 1][1]);
        }
        
        return dp[0][0];
    }
}

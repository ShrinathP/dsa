ðŸ§  Logic:
Keep track of current index idx.
For each element in preorder:
If it fits within (min, max) bounds â†’ create node.
Recur for left (max = current val) and right (min = current val) subtrees.
*/

class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int x) { val = x; }
}

class Solution {
    int idx = 0;

    public TreeNode bstFromPreorder(int[] preorder) {
        return build(preorder, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }

    private TreeNode build(int[] preorder, int min, int max) {
        if (idx >= preorder.length) return null;

        int val = preorder[idx];
        if (val < min || val > max) return null;

        TreeNode root = new TreeNode(val);
        idx++;
        root.left = build(preorder, min, val);
        root.right = build(preorder, val, max);

        return root;
    }
}
